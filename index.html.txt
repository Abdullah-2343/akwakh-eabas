        // Hero Slideshow Functions
        function loadHeroSlideshow() {
            const slideshow = document.getElementById('heroSlideshow');
            const indicators = document.getElementById('heroIndicators');
            
            slideshow.innerHTML = '';
            indicators.innerHTML = '';
            
            heroImages.forEach((item, index) => {
                // Create slide
                const slide = document.createElement('div');
                slide.className = `hero-slide ${index === 0 ? 'active' : ''}`;
                
                if (item.type === 'video') {
                    slide.innerHTML = `
                        <video autoplay muted loop style="width: 100%; height: 100%; object-fit: cover;">
                            <source src="${item.url}" type="video/mp4">
                            <!-- Fallback to poster image if video fails -->
                        </video>
                    `;
                    slide.style.backgroundImage = `url('${item.poster}')`;
                } else {
                    slide.style.backgroundImage = `url('${item.url}')`;
                }
                
                slideshow.appendChild(slide);
                
                // Create indicator
                const indicator = document.createElement('div');
                indicator.className = `hero-indicator ${index === 0 ? 'active' : ''}`;
                indicator.title = item.title;
                indicator.onclick = () => goToSlide(index);
                
                // Add video icon for video slides
                if (item.type === 'video') {
                    indicator.innerHTML = '<span style="font-size: 8px;">▶</span>';
                }
                
                indicators.appendChild(indicator);
            });

            // Add hover pause functionality
            const heroSection = document.querySelector('.hero');
            heroSection.addEventListener('mouseenter', stopSlideshow);
            heroSection.addEventListener('mouseleave', startSlideshow);
        }

        function startSlideshow() {
            if (!autoPlayEnabled) return;
            stopSlideshow(); // Clear any existing interval
            slideInterval = setInterval(nextSlide, slideTimer);
        }

        function stopSlideshow() {
            if (slideInterval) {
                clearInterval(slideInterval);
            }
        }

        function updateSlideTimer(seconds) {
            slideTimer = seconds * 1000;
            if (autoPlayEnabled) {
                stopSlideshow();
                startSlideshow();
            }
        }

        function toggleAutoPlay(enabled) {
            autoPlayEnabled = enabled;
            if (enabled) {
                startSlideshow();
            } else {
                stopSlideshow();
            }
        }

        function nextSlide() {
            currentSlide = (currentSlide + 1) % heroImages.length;
            updateSlide();
        }

        function prevSlide() {
            currentSlide = (currentSlide - 1 + heroImages.length) % heroImages.length;
            updateSlide();
        }

        function goToSlide(index) {
            currentSlide = index;
            updateSlide();
            stopSlideshow();
            startSlideshow(); // Restart timer
        }

        function updateSlide() {
            const slides = document.querySelectorAll('.hero-slide');
            const indicators = document.querySelectorAll('.hero-indicator');
            
            slides.forEach((slide, index) => {
                slide.classList.toggle('active', index === currentSlide);
            });
            
            indicators.forEach((indicator, index) => {
                indicator.classList.toggle('active', index === currentSlide);
            });
        }

        // Load Footer
        function loadFooter() {
            const footer = document.getElementById('dynamicFooter');
            if (!footer) {
                console.error('Footer element not found');
                return;
            }
            
            // Clear existing content first
            footer.innerHTML = '';
            
            let footerContent = '<div class="container"><div style="display: grid; grid-template-columns: repeat(auto-fit, minmax(250px, 1fr)); gap: 2rem; margin-bottom: 2rem;">';
            
            // Loop through all footer sections
            Object.keys(footerData).forEach(key => {
                const section = footerData[key];
                
                if (key === 'copyright') return; // Skip copyright as it's handled separately
                
                if (section && section.active) {
                    if (key === 'mainSection') {
                        footerContent += `
                            <div>
                                <h3 style="margin-bottom: 1rem;">${section.title}</h3>
                                <p style="line-height: 1.6; color: #c8e6c9;">${section.description}</p>
                            </div>
                        `;
                    } else if (section.items && Array.isArray(section.items)) {
                        footerContent += `
                            <div>
                                <h4 style="margin-bottom: 1rem;">${section.title}</h4>
                                <p style="line-height: 1.8; color: #c8e6c9;">
                                    ${section.items.join('<br>')}
                                </p>
                            </div>
                        `;

            });
            
            footerContent += `
                </div>
                <div style="border-top: 1px solid #4a7c59; padding-top: 1rem; text-align: center; color: #c8e6c9;">
                    <p>${footerData.copyright}</p>
                </div>
            </div>`;
            
            footer.innerHTML = footerContent;
        }

        function setupSmoothScrolling() {
            document.querySelectorAll('a[href^="#"]').forEach(anchor => {
                anchor.addEventListener('click', function (e) {
                    e.preventDefault();
                    const target = document.querySelector(this.getAttribute('href'));
                    if (target) {
                        target.scrollIntoView({
                            behavior: 'smooth',
                            block: 'start'
                        });
                    }
                });
            });
        }

        function uploadHeroMedia() {
            const fileInput = document.getElementById('heroMediaUpload');
            const titleInput = document.getElementById('mediaTitle');
            const files = fileInput.files;
            
            if (files.length === 0) {
                alert('يرجى اختيار ملفات للرفع');
                return;
            }
            
            let addedCount = 0;
            
            Array.from(files).forEach((file, index) => {
                const url = URL.createObjectURL(file);
                const isVideo = file.type.startsWith('video/');
                let title = titleInput.value || file.name.replace(/\.[^/.]+$/, "");
                
                // Add number suffix if multiple files and custom title
                if (files.length > 1 && titleInput.value) {
                    title = `${titleInput.value} ${index + 1}`;
                }
                
                const newItem = {
                    type: isVideo ? 'video' : 'image',
                    url: url,
                    title: title
                };
                
                if (isVideo) {
                    // For videos, create a poster image
                    newItem.poster = 'data:image/svg+xml,<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 1200 600"><rect fill="%23333" width="1200" height="600"/><polygon fill="white" points="480,200 480,400 720,300" opacity="0.8"/><text x="600" y="500" text-anchor="middle" fill="white" font-size="24">' + title + '</text></svg>';
                }
                
                heroImages.push(newItem);
                addedCount++;
            });
            
            loadHeroSlideshow();
            loadAdminHeroImages();
            fileInput.value = '';
            titleInput.value = '';
            
            // Reset slideshow
            currentSlide = 0;
            stopSlideshow();
            if (autoPlayEnabled) startSlideshow();
            
            alert(`تم رفع ${addedCount} ملف بنجاح! ${addedCount === 1 ? '' : 'الآن لديك ' + heroImages.length + ' عنصر في المعرض الرئيسي.'}`);
        }

        function editHeroItem(index) {
            const item = heroImages[index];
            const newTitle = prompt('تعديل العنوان:', item.title);
            if (newTitle && newTitle.trim() !== '') {
                heroImages[index].title = newTitle.trim();
                loadAdminHeroImages();
                alert('تم تحديث العنوان بنجاح!');
            }
        }

        function moveHeroItem(index, direction) {
            if (direction === 'up' && index > 0) {
                [heroImages[index], heroImages[index - 1]] = [heroImages[index - 1], heroImages[index]];
                if (currentSlide === index) currentSlide = index - 1;
                else if (currentSlide === index - 1) currentSlide = index;
            } else if (direction === 'down' && index < heroImages.length - 1) {
                [heroImages[index], heroImages[index + 1]] = [heroImages[index + 1], heroImages[index]];
                if (currentSlide === index) currentSlide = index + 1;
                else if (currentSlide === index + 1) currentSlide = index;
            }
            
            loadHeroSlideshow();
            loadAdminHeroImages();
            updateSlide();
        }

        function deleteHeroImage(index) {
            if (confirm('هل أنت متأكد من حذف هذه الصورة الرئيسية؟')) {
                heroImages.splice(index, 1);

                        title: 'الصورة الافتراضية'
                    });
                }
                
                // Reset current slide if it's out of bounds
                if (currentSlide >= heroImages.length) {
                    currentSlide = 0;
                }
                
                loadHeroSlideshow();
                loadAdminHeroImages();
                stopSlideshow();
                startSlideshow();
            }
        }

                            ${comment.approved ? 'معتمد' : 'اعتماد'}
                        </button>
                        <button onclick="deleteComment(${comment.id})" 
                                style="background: #dc3545; color: white; border: none; padding: 5px 15px; border-radius: 5px;">
                            حذف
                        </button>
                    </div>
                </div>
            `).join('');
        }

        function updateBookingStatus(bookingId, status) {
            const booking = bookings.find(b => b.id === bookingId);
            if (booking) {
                booking.status = status;
                loadAdminBookings();
            }
        }

        function approveComment(commentId) {
            const comment = comments.find(c => c.id === commentId);
            if (comment) {
                comment.approved = true;
                loadComments();
                loadAdminComments();
            }
        }

        function deleteComment(commentId) {
            if (confirm('هل أنت متأكد من حذف هذا التعليق؟')) {
                comments = comments.filter(c => c.id !== commentId);
                loadComments();
                loadAdminComments();
            }
        }

        function uploadMedia() {
            const fileInput = document.getElementById('mediaUpload');
            const files = fileInput.files;
            
            if (files.length === 0) {
                alert('يرجى اختيار ملفات للرفع');
                return;
            }
            
            // Simulate file upload
            Array.from(files).forEach(file => {
                const url = URL.createObjectURL(file);
                galleryItems.push({
                    type: file.type.startsWith('image/') ? 'image' : 'video',
                    url: url,
                    title: file.name
                });
            });
            
            loadGallery();
            fileInput.value = '';
            alert('تم رفع الملفات بنجاح!');
        }
